001    package suncertify.db;
002    
003    /**
004     * A <code>SecurityException</code> class of exception that must be thrown when
005     * user tries to modify or delete record and uses invalid locking cookie. There
006     * are two reasons in that case:
007     * <ul>
008     * <li>User not locked record, calling previously locking routine</li>
009     * <li>Locking cookie that user uses is not a valid actual cookie</li>
010     * </ul>
011     * 
012     * @author Petr Shilkin
013     * @version 1.0
014     */
015    public class SecurityException extends Exception {
016    
017            /**
018             * A version number for this class so that serialization can occur without
019             * worrying about the underlying class changing between serialization and
020             * deserialization.
021             */
022            private static final long serialVersionUID = 5165L;
023    
024            /**
025             * Default constructor creates new instance of <code>SecurityException</code>
026             */
027            public SecurityException() {
028    
029            }
030    
031            /**
032             * Constructor creates new instance of <code>SecurityException</code>.
033             * 
034             * @param description
035             *         description of the exception, points to the cause of exception
036             */
037            public SecurityException(String description) {
038                    super(description);
039            }
040    }
