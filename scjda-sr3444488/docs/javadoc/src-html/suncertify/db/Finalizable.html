001    package suncertify.db;
002    
003    /**
004     * Interface <code>Finalizable</code> must implemented by classes that have
005     * clean up functionality. <br>
006     * In some cases object that have running threads inside could not be finalized.
007     * This interface tells that object must be forcedly finalized before all
008     * references to the one are lost.
009     */
010    public interface Finalizable {
011            
012            /**
013             * This method should be called to perform clean up.
014             */
015            void finalize();
016    }
