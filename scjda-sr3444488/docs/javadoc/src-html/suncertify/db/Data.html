018    /**
019     * Class <code>Data</code> implements interface {@link DBAccess} provides
020     * methods to operate a database file on local hard drive.
021     * 
022     * @author Petr Shilkin
023     * @version 1.0
024     */
025    public class Data implements DBAccess {
026            /**
027             * "field name" size, length in the header of file for each record
028             */
029            private static int fn_len = 1;
030            /**
031             * "field size" size, length in the header of file for each record
032             */
033            private static int fl_len = 1;
034    
035            /**
036             * "number of fields" size, length in the header of file for entire database
037             * file
038             */
039            private static final int nof_len = 2;
040    
041            /**
042             * magic cookie value captioned from database file
043             */
044            private static final byte[] mc_value = new byte[] { 0x00, 0x00, 0x02, 0x03 };
045    
046            /**
047             * character encoding of database file
048             */
049            private final static String encoding = "US-ASCII";
050            /**
051             * defined number of first record in database file
052             */
053            protected final static int FIRST_RECNO = 0;
054            /**
055             * space byte for fields that have not enough length
056             */
057            private final static byte space_byte = (byte) 0x20;
058            /*
059             * calculating length of record
060             */
061            static {
062                    int temp = 0;
063                    for (int fl : FIELD_LENGTH) {
064                            temp += fl;
065                    }
066                    record_length = temp;
067            }
068    
069            /*
070             * calculating header size of database file
071             */
072            static {
073                    int temp = 0;
074                    temp += mc_value.length;
075                    temp += nof_len;
076                    for (String fn : FIELD_SEQUENCE) {
077                            temp += fn_len;
078                            try {
079                                    temp += fn.getBytes(encoding).length;
080                            } catch (UnsupportedEncodingException e) {
081                            }
082                            temp += fl_len;
083    
084                    }
085                    header_offset = temp;
086            }

088            /**
089             * logger is for debugging purposes
090             */
091            private Logger log = Logger.getLogger(this.getClass().getPackage().getName());
092            /**
093             * set of record numbers of the deleted records
094             */
095            private final Set<Long> deleted = new HashSet<Long>();
096            /**
097             * deleted flag byte value
098             */
099            private final static byte deletedrecbyte = (byte) 0xFF;
100            /**
101             * random access file reference to operate database file
102             */
103            private RandomAccessFile file;
104            /**
105             * "deleted flag" size, length for each record
106             */
107            private final static int flaglength = 1;
108            /**
109             * data section offset, length between beginning on database file and first
110             * record
111             */
112            private static int header_offset;
113            /**
114             * Map of locks for locked record numbers. Key of map is unique object for
115             * record number, value is locking descriptor ("cookie").
116             */
117            private final Map<Long, Long> locks = new ConcurrentHashMap<Long, Long>();
118            /**
119             * Cache of unique objects that is used in the functionality
120             * "one lock"-"one record". Unique objects needed to perform valid blocking on
121             * the same record numbers. Using this cache allows to lock/unlock records
122             * separately each from other.
123             */
124            private final WeakHashSet<Long> numberCache = new WeakHashSet<Long>();
125            /**
126             * reference to helper object for file reading/writing
127             */
128            private final ReaderWriter readerwriter = new ReaderWriter();
129            /**
130             * record length without "deleted flag"
131             */
132            private final static int record_length;
133            /**
134             * Read/write blocking for <code>valid</code> and <code>deleted</code> sets.
135             * Many clients can read these sets but only one at time can write to sets.
136             * While writing is in progress reading can not be done.
137             */
138            private final ReentrantReadWriteLock rwlock = new ReentrantReadWriteLock();
139            /**
140             * Valid, i.e. not deleted, record numbers set
141             */
142            private final Set<Long> valid = new HashSet<Long>();
143    
144            /**
145             * valid flag byte value
146             */
147            private final static byte validrecbyte = 0x00;
148    
149            /**
150             * amount of all (deleted and valid) records in database file
151             */
152            private long recordCount;
153    
154            /**
155             * Constructor creates database operating object to operate database file on
156             * local hard drive.
157             * 
158             * @param path
159             *         local path to a database file
160             * @throws IOException
161             *          on read or write error
162             */
163            public Data(String path) throws IOException {
164    
165                    File file = new File(path);
166                    this.file = new RandomAccessFile(file, "rw");
167                    try {
168                            readerwriter.checkheader();
169                    } catch (IOException e) {
170                            log.log(Level.SEVERE, "wrong data file format");
171                            throw e;
172                    }
173                    buildIndex();
174            }
175    
176            /**
177             * {@inheritDoc}
178             */
179            public long createRecord(String[] data) {
180                    rwlock.writeLock().lock();
181                    try {
182                            Iterator<Long> it = deleted.iterator();
183                            boolean reuse;
184                            Long recNo;
185                            if (reuse = it.hasNext()) {
186                                    recNo = it.next();
187                            } else {
188                                    recNo = recordCount;
189                            }
190                            long pos = getRecPos(recNo);
191                            readerwriter.write(pos, validrecbyte);
192                            readerwriter.write(pos + flaglength, data);
193                            valid.add(recNo);
194                            if (reuse) {
195                                    deleted.remove(recNo);
196                            } else {
197                                    recordCount++;
198                            }
199                            return recNo;
200                    } catch (IOException e) {
201                            throw new RuntimeException("error writing file", e);
202                    } finally {
203                            rwlock.writeLock().unlock();
204                    }
205            }
206    
207            /**
208             * {@inheritDoc}
209             */
210            public void deleteRecord(long recNo, long lockCookie)
211                            throws RecordNotFoundException, SecurityException {
212                    Object recNoMutex = getMutexForRecNo(recNo);
213                    checkLockCookie(recNo, recNoMutex, lockCookie);
214                    rwlock.writeLock().lock();
215                    try {
216                            if (!valid.contains(recNo))
217                                    throw new RecordNotFoundException();
218                            long pos = getRecPos(recNo);
219                            readerwriter.write(pos, deletedrecbyte);
220                            valid.remove(recNo);
221                            deleted.add(recNo);
222                    } catch (IOException e) {
223                            throw new RuntimeException("error writing file", e);
224                    } finally {
225                            rwlock.writeLock().unlock();
226                    }
227            }
228    
229            /**
230             * Performs clean-up of unused object.
231             */
232            protected void finalize() {
233                    close();
234            }
235    
236            /**
237             * {@inheritDoc}
238             */
239            public long[] findByCriteria(String[] criteria) {
240                    long[] result = new long[valid.size()];
241                    int count = 0;
242                    for (Long recNo : valid) {
243                            try {
244                                    String[] data = readRecord(recNo);
245                                    if (checkCriteria(data, criteria)) {
246                                            result[count] = recNo;
247                                            count++;
248                                    }
249                            } catch (RecordNotFoundException e) {
250                                    log.log(Level.SEVERE, "record was valid but now not found");
251                            }
252                    }
253                    long[] cutresult = new long[count];
254                    System.arraycopy(result, 0, cutresult, 0, count);
255                    return cutresult;
256            }
257    
258            /**
259             * {@inheritDoc}
260             */
261            public long lockRecord(long recNo) throws RecordNotFoundException {
262                    log.log(Level.SEVERE, "lock record: " + recNo);
263                    Object recNoMutex = getMutexForRecNo(recNo);
264                    long lockCookie = 0;
265                    synchronized (recNoMutex) {
266                            while (locks.containsKey(recNo)) {
267                                    try {
268                                            recNoMutex.wait();
269                                    } catch (InterruptedException e) {
270                                            throw new RuntimeException(e);
271                                    }
272                            }
273                            rwlock.readLock().lock();
274                            try {
275                                    if (!valid.contains(recNo))
276                                            throw new RecordNotFoundException("invalid record number " + recNo);
277                                    lockCookie = System.nanoTime();
278                                    locks.put(recNo, lockCookie);
279                            } finally {
280                                    rwlock.readLock().unlock();
281                            }
282                    }
283                    log
284                                    .log(Level.SEVERE, "record " + recNo + " locked with cookie " + lockCookie);
285                    return lockCookie;
286    
287            }
288    
289            /**
290             * {@inheritDoc}
291             */
292            public String[] readRecord(long recNo) throws RecordNotFoundException {
293                    rwlock.readLock().lock();
294                    try {
295                            if (!valid.contains(recNo))
296                                    throw new RecordNotFoundException();
297                            long pos = getRecPos(recNo);
298                            String[] data = readerwriter.readData(pos + flaglength);
299                            return data;
300                    } catch (IOException e) {
301                            throw new RuntimeException(e);
302                    } finally {
303                            rwlock.readLock().unlock();
304                    }
305            }
306    
307            /**
308             * {@inheritDoc}
309             */
310            public void unlock(long recNo, long cookie) throws SecurityException {
311                    log.log(Level.SEVERE, "unlock: " + recNo + ", " + cookie);
312                    Object recNoMutex = getMutexForRecNo(recNo);
313                    synchronized (recNoMutex) {
314                            Long lock = locks.get(recNo);
315                            if (lock != null && lock.longValue() == cookie) {
316                                    locks.remove(recNo);
317                                    recNoMutex.notifyAll();
318                            } else {
319                                    throw new SecurityException("invalid unlock cookie");
320                            }
321                    }
322            }
323    
324            /**
325             * {@inheritDoc}
326             */
327            public void updateRecord(long recNo, String[] data, long lockCookie)
328                            throws RecordNotFoundException, SecurityException {
329                    log.log(Level.SEVERE, "update: " + recNo + ", " + lockCookie);
330                    Object recNoMutex = getMutexForRecNo(recNo);
331                    checkLockCookie(recNo, recNoMutex, lockCookie);
332                    rwlock.readLock().lock();
333                    try {
334                            if (!valid.contains(recNo))
335                                    throw new RecordNotFoundException("invalid record number");
336                    } finally {
337                            rwlock.readLock().unlock();
338                    }
339    
340                    long pos = getRecPos(recNo);
341                    try {
342                            readerwriter.write(pos + flaglength, data);
343                    } catch (IOException e) {
344                            throw new RuntimeException(e);
345                    }
346            }
347    
348            /**
349             * Method fills <code>valid</code> and <code>deleted</code> sets with values,
350             * according data in database file.
351             * 
352             * @throws IOException
353             *          on file reading error
354             */
355            private void buildIndex() throws IOException {
356                    rwlock.writeLock().lock();
357                    try {
358                            valid.clear();
359                            deleted.clear();
360                            for (long pos = header_offset, recNo = FIRST_RECNO; pos < file.length(); pos += (record_length + flaglength), recNo++) {
361                                    byte b = readerwriter.read(pos);
362                                    if (b == validrecbyte) {
363                                            valid.add(recNo);
364                                    } else if (b == deletedrecbyte) {
365                                            deleted.add(recNo);
366                                    } else {
367                                            throw new RuntimeException("flag expected but other found: " + b);
368                                    }
369                            }
370                            recordCount = valid.size() + deleted.size();
371                    } finally {
372                            rwlock.writeLock().unlock();
373                    }
374    
375            }

377            /**
378             * Method checks if record is accepted by filter criteria.
379             * 
380             * @param dataArr
381             *         record fields
382             * @param criteriaArr
383             *         filter criteria
384             * @return <code>true</code> if record is accepted, <code>false</code> if
385             *         record is not accepted
386             */
387            private boolean checkCriteria(String[] dataArr, String[] criteriaArr) {
388                    try {
389                            for (int i = 0; i < FIELD_SEQUENCE.length; i++) {
390                                    String data = dataArr[i];
391                                    String criteria = criteriaArr[i];
392                                    if (criteria == null || data == null)
393                                            continue;
394                                    if (!data.startsWith(criteria))
395                                            return false;
396                            }
397                    } catch (RuntimeException e) {
398                            return false;
399                    }
400                    return true;
401            }
402    
403            /**
404             * Validates locking descriptor for specified record.
405             * 
406             * @param recNoRef
407             *         "unique per record number" object
408             * @param lockCookie
409             *         locking descriptor
410             * @throws SecurityException
411             *          if locking descriptor is invalid
412             */
413            private void checkLockCookie(Long recNo, Object recNoMutex, long lockCookie)
414                            throws SecurityException {
415                    synchronized (recNoMutex) {
416                            log.log(Level.SEVERE, "check lock: " + recNo + ", " + lockCookie);
417                            Long lock = locks.get(recNo);
418                            if (lock == null || lock.longValue() != lockCookie)
419                                    throw new SecurityException("invalid cookie passed");
420                    }
421            }

423            /**
424             * Closes this random access file stream and releases any system resources
425             * associated with the stream when garbage collection determines that there are
426             * no more references to the object.
427             */
428            private void close() {
429                    if (file != null)
430                            try {
431                                    file.close();
432                                    file = null;
433                            } catch (IOException e) {
434                                    log.log(Level.SEVERE, "Error closing file", e);
435                            }
436            }

438            /**
439             * Returns unique object for record number. Unique object used to gain access
440             * to the record index in sets: <code>valid, deleted</code>. Also it provides
441             * ability to lock and unlock records independently each from other.
442             * 
443             * @param recNo
444             *         record number
445             * @return "unique per record number" object
446             */
447            private Object getMutexForRecNo(long recNo) {
448                    Long number = numberCache.get(recNo);
449                    if (number == null) {
450                            number = new Long(recNo);
451                            numberCache.add(number);
452                    }
453                    return number;
454            }
455    
456            /**
457             * Returns position of record in file.
458             * 
459             * @param recNo
460             *         record number
461             * @return position of record in database file
462             */
463            private long getRecPos(long recNo) {
464                    return header_offset + (recNo - FIRST_RECNO) * (record_length + flaglength);
465            }

467            /**
468             * Helper class for reading and writing random access file that is database
469             * file.
470             */
471            private class ReaderWriter {
472    
473                    /**
474                     * Checks if existing header of file is valid
475                     * 
476                     * @throws IOException
477                     *          on file reading error or if header is not valid
478                     */
479                    public void checkheader() throws IOException {
480                            if (file.length() < header_offset) {
481                                    throw new IOException("database file is too small to be valid");
482                            }
483                            synchronized (file) {
484                                    file.seek(0);
485                                    byte[] hbytes = new byte[mc_value.length];
486                                    file.read(hbytes);
487                                    if (!Arrays.equals(hbytes, mc_value))
488                                            throw new IOException("wrong database file format");
489                            }
490                    }
491    
492                    /**
493                     * Reads one byte in position <code>pos</code> of random access file
494                     * 
495                     * @param pos
496                     *         position of byte
497                     * @return byte method reads
498                     * @throws IOException
499                     *          on error during reading
500                     */
501                    public byte read(long pos) throws IOException {
502                            byte[] b = new byte[1];
503                            synchronized (file) {
504                                    file.seek(pos);
505                                    file.read(b);
506                            }
507                            return b[0];
508                    }
509    
510                    /**
511                     * Returns field values based on database record scheme as array of String
512                     * 
513                     * @param pos
514                     *         position of data in random access file
515                     * @param fields
516                     *         database record scheme
517                     * @return array of String representing fields of record
518                     * @throws IOException
519                     *          on file reading error
520                     */
521                    public String[] readData(long pos) throws IOException {
522                            String[] result = new String[FIELD_SEQUENCE.length];
523                            synchronized (file) {
524                                    file.seek(pos);
525                                    for (int i = 0; i < FIELD_LENGTH.length; i++) {
526                                            byte[] fielddata = new byte[FIELD_LENGTH[i]];
527                                            file.read(fielddata, 0, fielddata.length);
528                                            result[i] = new String(fielddata, encoding).trim();
529                                    }
530                            }
531                            return result;
532                    }
533    
534                    /**
535                     * Writes only one byte to a specified position <code>pos</code> in the random
536                     * access file
537                     * 
538                     * @param pos
539                     *         position in random access file
540                     * @param b
541                     *         one byte to write
542                     * @throws IOException
543                     *          on file writing error
544                     */
545                    public void write(long pos, byte b) throws IOException {
546                            synchronized (file) {
547                                    file.seek(pos);
548                                    file.write(b);
549                            }
550                    }
551    
552                    /**
553                     * Writes data to a specified position <code>pos</code> in the random access
554                     * file, according database record scheme.
555                     * 
556                     * @param pos
557                     *         position in a random access file
558                     * @param data
559                     *         array of String, representing fields of record
560                     * @throws IOException
561                     *          on file writing error
562                     */
563                    public void write(long pos, String[] data) throws IOException {
564                            synchronized (file) {
565                                    file.seek(pos);
566                                    for (int i = 0; i < FIELD_LENGTH.length; i++) {
567                                            byte[] fielddata = new byte[FIELD_LENGTH[i]];
568                                            byte[] outdata = data[i].getBytes(encoding);
569                                            if (fielddata.length > outdata.length)
570                                                    Arrays.fill(fielddata, space_byte);
571                                            System.arraycopy(outdata, 0, fielddata, 0, Math.min(fielddata.length,
572                                                            outdata.length));
573                                            file.write(fielddata, 0, fielddata.length);
574                                    }
575                            }
576                    }
577            }
