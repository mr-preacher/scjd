001    package suncertify.sockets;
002    
003    import java.io.Serializable;
004    import java.util.Arrays;
005    
006    /**
007     * Class <code>ResponseResult</code> purposed for data transfer between
008     * {@link RecordSocketServer} and {@link RecordSocketClient}.<br>
009     * It wraps transferred data object of response, which can be one of:
010     * <ul>
011     * <li>Exception, that may occur on the server side</li>
012     * <li>Array of long, that used to represent numbers of the found records</li>
013     * <li>Array of String, that used to represent record fields</li>
014     * <li>Long, that used to represent record number</li>
015     * </ul>
016     * 
017     * @author Petr Shilkin
018     * @version 1.0
019     */
020    public class ResponseResult implements Serializable {
021            /**
022             * A version number for this class so that serialization can occur without
023             * worrying about the underlying class changing between serialization and
024             * deserialization.
025             */
026            private static final long serialVersionUID = 5165L;
027    
028            /**
029             * An internal reference to a transferring data object.
030             */
031            private Object data = null;
032    
033            /**
034             * Constructor of <code>ResponseResult</code> object.<br>
035             * 
036             * @param o
037             *         transferring data object to wrap
038             */
039            public ResponseResult(Object o) {
040                    if (!(o instanceof Exception || o instanceof long[] || o instanceof String[] || o instanceof Long)) {
041                            throw new IllegalArgumentException(
042                                            "incompatible transferring data object passed");
043                    }
044                    this.data = o;
045            }
046    
047            /**
048             * Return the array of long as transfered data.
049             * 
050             * @return array of long if transferred object is array of long
051             */
052            public long[] getLongArray() {
053                    return (long[]) data;
054            }
055    
056            /**
057             * Returns the array of String as transfered data.
058             * 
059             * @return array of String if transferred object is array of String
060             */
061            public String[] getStringArray() {
062                    return (String[]) data;
063            }
064    
065            /**
066             * Returns long as a transferred data.
067             * 
068             * @return long object if transfered object is {@link Long}
069             */
070            public long getLong() {
071                    return (Long) data;
072            }
073    
074            /**
075             * Returns exception as a transferred data.
076             * 
077             * @return exception object if transferred object is {@link Exception}
078             */
079            public Exception getException() {
080                    return (Exception) this.data;
081            }
082    
083            /**
084             * Returns <code>true</code> if transferred data is {@link Exception}
085             * 
086             * @return <code>true</code> if transferred object is exception,
087             *         <code>false</code> if transferred object is not exception
088             */
089            public boolean isException() {
090                    return data instanceof Exception;
091            }
092    
093            /**
094             * Creates a string representing this result for debugging and logging
095             * purposes.
096             * 
097             * @return a string representing this {@link ResponseResult}
098             */
099            public String toString() {
100                    if (data instanceof Object[]) {
101                            return Arrays.toString((Object[]) data);
102                    } else if (data != null) {
103                            return data.toString();
104                    }
105                    return "null";
106            }
107    }
