001    package suncertify.sockets;
002    
003    /**
004     * The enumerated list of possible commands we can send from the client to the
005     * server.<br>
006     * Each of the commands represents a certain action for database, that is
007     * operated remotely:
008     * <ul>
009     * <li>Find - finds matching records</li>
010     * <li>Add - creates and persists new record</li>
011     * <li>Delete - deletes the specified record</li>
012     * <li>Modify - changes field of the specified record</li>
013     * <li>Read - reads the specified record</li>
014     * <li>Lock - locks the specified record for changing</li>
015     * <li>Unlock - unlocks the specified record after locking</li>
016     * </ul>
017     * 
018     * @author Petr Shilkin
019     * @version 1.0
020     */
021    public enum SocketCommand {
022            /** indicates that the command object has not been set. */
023            UNSPECIFIED,
024            /** request will be performing a Find action. */
025            FIND,
026            /** request will be performing a Create action. */
027            ADD,
028            /** request will be performing a Delete action. */
029            DELETE,
030            /** request will be performing a Update action. */
031            MODIFY,
032            /** request will be performing a Lock action. */
033            LOCK,
034            /** request will be performing a Unlock action. */
035            UNLOCK,
036            /** request will be performing a Read action. */
037            READ
038    }
