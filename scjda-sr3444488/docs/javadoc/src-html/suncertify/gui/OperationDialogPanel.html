151            /**
152             * Constructor of <code>OperationDialogPanel</code> object.<br>
153             * Some text edit controls can be displayed or not, it depends on the type of
154             * operation:
155             * <ul>
156             * <li><code>OperationType.FIND</code> displays all text edit controls and
157             * check boxes for "exact" search functionality.</li>
158             * <li><code>OperationType.BOOK</code> displays only text edit control for
159             * owner field <code>owner</code>.</li>
160             * <li><code>OperationType.CREATE</code> and <code>OperationType.MODIFY</code>
161             * check boxes and all text edit controls except text edit control for owner
162             * field are not displayed.</li>
163             * </ul>
164             * 
165             * @param type
166             *         type of operation
167             */
168            public OperationDialogPanel(OperationType type) {
169                    initComponents();
170                    initMapping();
171                    initToolTips();
172                    switch (type) {
173                    case CREATE:
174                    case MODIFY:
175                            checkName.setVisible(false);
176                            checkLoc.setVisible(false);
177                            labOwnr.setVisible(false);
178                            fieldOwnr.setVisible(false);
179                            break;
180                    case BOOK:
181                            labName.setVisible(false);
182                            fieldName.setVisible(false);
183                            checkName.setVisible(false);
184                            labLoc.setVisible(false);
185                            fieldLoc.setVisible(false);
186                            checkLoc.setVisible(false);
187                            labSpec.setVisible(false);
188                            fieldSpec.setVisible(false);
189                            labSize.setVisible(false);
190                            fieldSize.setVisible(false);
191                            labRate.setVisible(false);
192                            fieldRate.setVisible(false);
193                            break;
194                    }
195            }

197            /**
198             * Returns text values of the text edit controls according to field names
199             * constants.
200             * 
201             * @param fieldNames
202             *         array of field names
203             * @return array of String values in text controls according to field names
204             */
205            public String[] getTextValues(String[] fieldNames) {
206                    String[] res = new String[fieldNames.length];
207                    for (int i = 0; i < fieldNames.length; i++) {
208                            String fieldName = fieldNames[i];
209                            JTextComponent comp = mapping.get(fieldName);
210                            if (comp == null)
211                                    continue;
212                            res[i] = comp.getText();
213    
214                    }
215                    return res;
216            }
217    
218            /**
219             * Sets text values to text edit controls according to field names array.
220             * 
221             * @param values
222             *         value to set to the text controls
223             * @param fieldNames
224             *         array of field names, defining sequence of values
225             */
226            public void setTextValues(String[] values, String[] fieldNames) {
227                    if (values.length != fieldNames.length)
228                            throw new IllegalArgumentException("invalid fieldname count ");
229                    for (int i = 0; i < fieldNames.length; i++) {
230                            String fieldName = fieldNames[i];
231                            JTextField comp = mapping.get(fieldName);
232                            if (comp == null)
233                                    continue;
234                            String value = values[i];
235                            if (value == null) {
236                                    value = "";
237                            } else
238                                    value = value.trim();
239                            comp.setText(value);
240                    }
241            }

252            /**
253             * Shows dialog containing text edit controls filled with <code>values</code>.<br>
254             * Array <code>fields</code> defines the sequence of values in the
255             * <code>values</code> array.<br>
256             * Dialog has two options to allow user commit or cancel possible changes.<br>
257             * 
258             * @param values
259             *         text values to display in text edit controls, elements of array can
260             *         be changed
261             * @param fields
262             *         field names defining the sequence of text values
263             * @param type
264             *         operation type of current operation
265             * @return <code>true</code> if user clicked OK to commit changes,
266             *         <code>false</code> if user is canceled changes
267             */
268            public static boolean showDialog(final String[] values, final String[] fields,
269                            OperationType type) {
270                    boolean[] b = new boolean[] { false, false };
271                    return showDialog(values, fields, b, type);
272            }

274            /**
275             * Shows dialog containing text edit controls filled with <code>values</code>.<br>
276             * Array <code>fields</code> defines the sequence of values in the
277             * <code>values</code> array.<br>
278             * Dialog has two options to allow user commit or cancel possible changes.<br>
279             * 
280             * @param values
281             *         text values to display in text edit controls, elements of array can
282             *         be changed
283             * @param fields
284             *         field names defining the sequence of text values
285             * @param check
286             *         check box values which define "exact" search conditions, elements of
287             *         array can be changed
288             * @param type
289             *         operation type for current operation
290             * @return <code>true</code> if user clicked OK to commit changes,
291             *         <code>false</code> if user is canceled changes
292             */
293            public static boolean showDialog(final String[] values, final String[] fields,
294                            final boolean[] check, OperationType type) {
295                    if (check == null || check.length != 2)
296                            throw new IllegalArgumentException(
297                                            "check must be not null. array of 2 boolean expected");
298                    OperationDialogPanel dialogPane = new OperationDialogPanel(type);
299                    String title = null;
300                    switch (type) {
301                    case BOOK:
302                            title = "Book";
303                            break;
304                    case FIND:
305                            title = "Find";
306                            break;
307                    case CREATE:
308                            title = "Create";
309                            break;
310                    case MODIFY:
311                            title = "Modify";
312                            break;
313                    }
314                    dialogPane.setTextValues(values, fields);
315                    CustomDialog cd = new CustomDialog(title, dialogPane);
316                    if (cd.isOKPressed()) {
317                            System.arraycopy(dialogPane.getTextValues(fields), 0, values, 0,
318                                            values.length);
319                            System.arraycopy(dialogPane.getFlagsValues(), 0, check, 0, check.length);
320                            return true;
321                    } else {
322                            return false;
323                    }
324            }
