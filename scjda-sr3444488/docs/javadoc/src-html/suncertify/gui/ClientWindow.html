382            /**
383             * Requests database for records in database that matches search criteria.
384             * 
385             * @param criteriaArr
386             *         search criteria array of string
387             * @param exChecks
388             *         exact search flags for name and location fields, can be
389             *         <code>null</code> if no exact search requested
390             * @return table model to use in {@link JTable} component
391             * @throws GuiException
392             *          on any error
393             */
394            private RecordTableModel findRecords(String[] criteriaArr, boolean[] exChecks)
395                            throws GuiException {
396                    try {
397                            RecordTableModel out = new RecordTableModel();
398                            long[] res = application.getDBAccess().findByCriteria(criteriaArr);
399                            nextRecNo: for (long recNo : res) {
400                                    try {
401                                            String[] dataArr = readRecord(recNo);
402                                            if (exChecks != null) {
403                                                    for (int i = 0; i < exChecks.length; i++) {
404                                                            String data = dataArr[i];
405                                                            if (exChecks[i] && !data.equals(criteriaArr[i])) {
406                                                                    continue nextRecNo;
407                                                            }
408                                                    }
409                                            }
410                                            out.addRecord(recNo, dataArr);
411                                    } catch (Exception e) {
412                                            log.log(Level.SEVERE, "record already not found " + recNo);
413                                    }
414                            }
415                            return out;
416                    } catch (Exception e) {
417                            throw new GuiException(e);
418                    }
419            }

546            /**
547             * Updates row selection in table after table model changed
548             * 
549             * @param rowNo
550             *         preferred row number
551             */
552            private void updateRowSelection(int rowNo) {
553                    int rcount = 0;
554                    if ((rcount = mainTable.getModel().getRowCount()) > 0) {
555                            if (rcount > rowNo) {
556                                    mainTable.getSelectionModel().setSelectionInterval(rowNo, rowNo);
557                            } else {
558                                    mainTable.getSelectionModel().setSelectionInterval(rcount - 1, rcount - 1);
559                            }
560                    }
561            }

566            /**
567             * ActionListener class for Book (Unbook) operation.
568             */
569            private class BookUnbookActionListener implements ActionListener {
570                    /**
571                     * {@inheritDoc}
572                     */
573                    public void actionPerformed(ActionEvent e) {
574                            int row;
575                            if ((row = mainTable.getSelectedRow()) != -1) {
576                                    row = mainTable.convertRowIndexToModel(row);
577                                    RecordTableModel tableData = (RecordTableModel) mainTable.getModel();
578                                    long recNo = tableData.getRecNoByRow(row);
579                                    long lockCookie = 0;
580                                    try {
581                                            int res = JOptionPane.showConfirmDialog(ClientWindow.this,
582                                                            "Do you really want to book/unbook record", "Are you sure?",
583                                                            JOptionPane.YES_NO_OPTION);
584                                            if (res == JOptionPane.YES_OPTION) {
585                                                    lockCookie = lockRecord(recNo);
586                                                    String[] data = readRecord(recNo);
587                                                    if (OperationDialogPanel.showDialog(data, DBAccess.FIELD_SEQUENCE,
588                                                                    OperationType.BOOK)) {
589                                                            updateRecord(recNo, data, lockCookie);
590                                                            tableData.update(row, readRecord(recNo));
591                                                    }
592                                            }
593                                    } catch (GuiException ex) {
594                                            application.handleException("Error occured when book/unbook record\n"
595                                                            + ex.getMessage());
596                                    } finally {
597                                            if (lockCookie != 0) {
598                                                    try {
599                                                            unlockRecord(recNo, lockCookie);
600                                                    } catch (GuiException ex) {
601                                                            application.handleException("Error occured when unlocking record\n"
602                                                                            + ex.getMessage());
603                                                    }
604                                            }
605                                    }
606                            }
607                    }
608            }
