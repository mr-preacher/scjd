001    package suncertify.gui.utils;
002    
003    import java.util.logging.Logger;
004    import java.util.regex.Matcher;
005    import java.util.regex.Pattern;
006    import javax.swing.JFormattedTextField;
007    
008    /**
009     * Class <code>JRegexTextField</code> extends ValidatingTextField. It is a
010     * control like {@link JFormattedTextField} but uses a regular expression
011     * pattern instead of mask pattern.
012     * 
013     * @author Petr Shilkin
014     * @version 1.0
015     */
016    public class JRegexTextField extends ValidatingTextField {
017            /**
018             * A version number for this class so that serialization can occur without
019             * worrying about the underlying class changing between serialization and
020             * deserialization.
021             */
022            private static final long serialVersionUID = 5165L;
023            /**
024             * Logger for debugging purposes
025             */
026            private transient Logger log = Logger.getLogger(this.getClass().getPackage()
027                            .getName());
028    
029            /**
030             * Constructor of <code>JRegexTextField</code>. Creates new edit control with a
031             * specified regular expression.
032             * 
033             * @param regex
034             *         regular expression used in this control
035             */
036            public JRegexTextField(final String regex) {
037                    super(new ValidationHandler() {
038                            /**
039                             * pattern to use for validation
040                             */
041                            private Pattern pattern = Pattern.compile(regex);
042    
043                            /**
044                             * {@inheritDoc}
045                             */
046                            public boolean validate(String text) {
047                                    Matcher matcher = pattern.matcher(text);
048                                    boolean result = matcher.matches();
049                                    return result;
050                            }
051                    });
052            }
053    
054    }
