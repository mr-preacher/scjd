001    package suncertify.gui.utils;
002    
003    /**
004     * Interface <code>ValidationHandler</code> purposed for the validation handling
005     * of text values.
006     * 
007     * @author Petr Shilkin
008     * @version 1.0
009     */
010    public interface ValidationHandler {
011            /**
012             * Method must return <code>true</code> if text value is validated, and
013             * <code>false</code> if validation of text value failed.
014             * 
015             * @param text
016             *         text value to be validated
017             * @return <code>true</code> if text value is validated, <code>false</code> if
018             *         validation failed
019             */
020            public boolean validate(String text);
021    
022    }
