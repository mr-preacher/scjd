001    package suncertify.gui;
002    
003    /**
004     * Class <code>GuiException</code> purposed to wrap all occurring exceptions,
005     * which thrown on the gui side of application.
006     * 
007     * @author Petr Shilkin
008     * @version 1.0
009     */
010    public class GuiException extends Exception {
011            /**
012             * A version number for this class so that serialization can occur without
013             * worrying about the underlying class changing between serialization and
014             * deserialization. Not that we ever serialize this class of course, but
015             * Exception implements Serializable, so therefore by default we do as well.
016             */
017            private static final long serialVersionUID = 5165L;
018    
019            /**
020             * Creates a default <code>GuiException</code> instance.
021             */
022            public GuiException() {
023            }
024    
025            /**
026             * Creates a <code>GuiException</code> instance and chains an exception.
027             * 
028             * @param e
029             *         The exception to wrap and chain.
030             */
031            public GuiException(Throwable e) {
032                    super(e);
033            }
034    }
