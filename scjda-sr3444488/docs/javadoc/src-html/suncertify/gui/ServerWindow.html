001    package suncertify.gui;
002    
003    import java.awt.event.ActionEvent;
004    import java.awt.event.ActionListener;
005    import java.awt.event.InputEvent;
006    import java.awt.event.KeyEvent;
007    import java.awt.event.WindowAdapter;
008    import java.awt.event.WindowEvent;
009    import java.util.logging.Handler;
010    import java.util.logging.Level;
011    import java.util.logging.LogRecord;
012    import java.util.logging.Logger;
013    import java.util.logging.SimpleFormatter;
014    import javax.swing.BorderFactory;
015    import javax.swing.JLabel;
016    import javax.swing.JMenu;
017    import javax.swing.JMenuBar;
018    import javax.swing.JMenuItem;
019    import javax.swing.JOptionPane;
020    import javax.swing.JPanel;
021    import javax.swing.JScrollPane;
022    import javax.swing.JSeparator;
023    import javax.swing.JTextArea;
024    import javax.swing.KeyStroke;
025    import javax.swing.SwingUtilities;
026    import javax.swing.WindowConstants;
027    import suncertify.Application;
028    import suncertify.ApplicationWindow;
029    
030    /**
031     * Class <code>ServerWindow</code> extends {@link ApplicationWindow}. This is a
032     * window that appears when application starts in the server mode.
033     * 
034     * @author Petr Shilkin
035     * @version 1.0
036     */
037    public class ServerWindow extends ApplicationWindow {
038            /**
039             * A version number for this class so that serialization can occur without
040             * worrying about the underlying class changing between serialization and
041             * deserialization.
042             */
043            private static final long serialVersionUID = 5165L;
044            /**
045             * Logger for debugging issues
046             */
047            private transient Logger log = Logger.getLogger(this.getClass().getPackage()
048                            .getName());
049    
050            /**
051             * about menu item
052             */
053            private JMenuItem aboutMI;
054            /**
055             * action menu
056             */
057            private JMenu actionMenu;
058            /**
059             * clear log menu item
060             */
061            private JMenuItem clearLogMI;
062            /**
063             * exit application menu item
064             */
065            private JMenuItem exitMI;
066            /**
067             * file menu
068             */
069            private JMenu fileMenu;
070            /**
071             * file menu separator
072             */
073            private JSeparator fileMenuSeparator;
074    
075            /**
076             * options menu item
077             */
078            private JMenuItem optionsMI;
079            /**
080             * help menu item
081             */
082            private JMenuItem helpMI;
083            /**
084             * help menu
085             */
086            private JMenu helpMenu;
087            /**
088             * main menu bar
089             */
090            private JMenuBar mainMenuBar;
091            /**
092             * log file text area
093             */
094            private JTextArea mainTextArea;
095            /**
096             * status label
097             */
098            private JLabel statusLabel;
099            /**
100             * status panel
101             */
102            private JPanel statusPane;
103            /**
104             * scroll pane for text area
105             */
106            private JScrollPane textScrollPane;
107    
108            /**
109             * Creates new <code>ServerWindow</code> object
110             * 
111             * @param application
112             *         reference to the application object
113             */
114            public ServerWindow(final Application application) {
115                    super(application);
116                    initComponents();
117                    initLoggerHandler();
118                    initOpenListener();
119            }
120    
121            /**
122             * Initializes listener to display option dialog on startup
123             */
124            private void initOpenListener() {
125                    this.addWindowListener(new WindowAdapter() {
126                            /**
127                             * Invokes option menu handler to show dialog with options i.e. settings
128                             */
129                            public void windowOpened(WindowEvent e) {
130                                    SwingUtilities.invokeLater(new Thread() {
131                                            public void run() {
132                                                    updateWindowContent();
133                                                    optionsMIActionPerformed();
134                                            }
135                                    });
136                            }
137                    });
138            }
139            
140            /**
141             * {@inheritDoc} <br>
142             * Changes status label text according server state when settings are changed.
143             */
144            public void updateWindowContent() {
145                    int port = application.getPortServerRunning();
146                    statusLabel.setText(port > -1 ? "server running on port " + port
147                                    : "server not running");
148            }
149    
150            /**
151             * Initializes logging handler to log into text area
152             */
153            private void initLoggerHandler() {
154                    ServerLoggerHandler handler = new ServerLoggerHandler();
155                    handler.setFormatter(new SimpleFormatter());
156                    handler.setLevel(Level.SEVERE);
157    
158                    Logger log = Logger.getLogger("suncertify.db");
159                    log.addHandler(handler);
160                    log = Logger.getLogger("suncertify.db.utils");
161                    log.setLevel(Level.OFF);
162                    log = Logger.getLogger("suncertify.sockets");
163                    log.addHandler(handler);
164    
165            }
166    
167            /**
168             * Initiates components of frame
169             */
170            private void initComponents() {
171                    statusPane = new JPanel();
172                    statusLabel = new JLabel();
173                    textScrollPane = new JScrollPane();
174                    mainTextArea = new JTextArea();
175                    mainMenuBar = new JMenuBar();
176                    fileMenu = new JMenu();
177                    optionsMI = new JMenuItem();
178                    fileMenuSeparator = new JSeparator();
179                    exitMI = new JMenuItem();
180                    actionMenu = new JMenu();
181                    clearLogMI = new JMenuItem();
182                    helpMenu = new JMenu();
183                    helpMI = new JMenuItem();
184                    aboutMI = new JMenuItem();
185                    setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
186                    statusPane.setPreferredSize(new java.awt.Dimension(400, 17));
187                    statusPane.setLayout(new java.awt.BorderLayout());
188                    statusLabel.setText("");
189                    statusLabel.setBorder(BorderFactory.createEmptyBorder(0, 7, 0, 0));
190                    statusPane.add(statusLabel, java.awt.BorderLayout.CENTER);
191                    getContentPane().add(statusPane, java.awt.BorderLayout.SOUTH);
192                    textScrollPane.setPreferredSize(new java.awt.Dimension(106, 200));
193                    mainTextArea.setColumns(20);
194                    mainTextArea.setRows(5);
195                    textScrollPane.setViewportView(mainTextArea);
196                    getContentPane().add(textScrollPane, java.awt.BorderLayout.CENTER);
197                    fileMenu.setText("File");
198                    optionsMI.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_O,
199                                    InputEvent.CTRL_MASK));
200                    optionsMI.setText("Options...");
201                    optionsMI.addActionListener(new ActionListener() {
202                            public void actionPerformed(ActionEvent evt) {
203                                    optionsMIActionPerformed();
204                            }
205                    });
206                    fileMenu.add(optionsMI);
207                    fileMenu.add(fileMenuSeparator);
208                    exitMI.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_F4,
209                                    InputEvent.ALT_MASK));
210                    exitMI.setText("Exit");
211                    exitMI.addActionListener(new ActionListener() {
212                            public void actionPerformed(ActionEvent evt) {
213                                    exitMIActionPerformed();
214                            }
215                    });
216                    fileMenu.add(exitMI);
217                    mainMenuBar.add(fileMenu);
218                    actionMenu.setText("Action");
219                    actionMenu.setActionCommand("Action");
220                    clearLogMI.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_C,
221                                    InputEvent.ALT_MASK | InputEvent.CTRL_MASK));
222                    clearLogMI.setText("Clear Log");
223                    clearLogMI.addActionListener(new ActionListener() {
224                            public void actionPerformed(ActionEvent evt) {
225                                    clearLogMIActionPerformed();
226                            }
227                    });
228                    actionMenu.add(clearLogMI);
229                    mainMenuBar.add(actionMenu);
230                    helpMenu.setText("Help");
231                    helpMI.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_F1, 0));
232                    helpMI.setText("Help");
233                    helpMI.addActionListener(new ActionListener() {
234                            public void actionPerformed(ActionEvent evt) {
235                                    helpMIActionPerformed();
236                            }
237                    });
238                    helpMenu.add(helpMI);
239                    aboutMI.setText("About");
240                    aboutMI.addActionListener(new ActionListener() {
241                            public void actionPerformed(ActionEvent evt) {
242                                    aboutMIActionPerformed();
243                            }
244                    });
245                    helpMenu.add(aboutMI);
246                    mainMenuBar.add(helpMenu);
247                    setJMenuBar(mainMenuBar);
248                    pack();
249            }
250    
251            /**
252             * Invokes settings dialog. Redefines settings for application.
253             */
254            private void optionsMIActionPerformed() {
255                    if (application.showOptionsDialog()) {
256                            application.cleanUp();
257                            application.applyOptions();
258                    }
259            }
260    
261            /**
262             * Clears text area from log messages.
263             */
264            private void clearLog() {
265                    mainTextArea.setText("");
266            }
267    
268            /**
269             * Adds log message to the text area.
270             * 
271             * @param msg
272             *         log message
273             */
274            private void addLogMsg(String msg) {
275                    mainTextArea.append(msg);
276            }
277    
278            /**
279             * Exits application, invoked from action listener of menu item.
280             */
281            private void exitMIActionPerformed() {
282                    this.close();
283            }
284    
285            /**
286             * Clears text area from logging, invoked from action listener of menu item.
287             */
288            private void clearLogMIActionPerformed() {
289                    clearLog();
290            }
291    
292            /**
293             * Shows help file.
294             */
295            private void helpMIActionPerformed() {
296                    String fileName = "docs/user_guide/server.htm";
297                    JOptionPane.showMessageDialog(this, "See help file at " + fileName, "Help",
298                                    JOptionPane.INFORMATION_MESSAGE);
299            }
300    
301            /**
302             * Shows about dialog.
303             */
304            private void aboutMIActionPerformed() {
305                    JOptionPane.showMessageDialog(this,
306                                    "Server Bodgitt and Scarper, LLC.\nPetr Shilkin 2009", "About",
307                                    JOptionPane.INFORMATION_MESSAGE);
308            }
309    
310            /**
311             * Server logging handler for logging into text area.
312             */
313            private class ServerLoggerHandler extends Handler {
314    
315                    /**
316                     * {@inheritDoc}<br>
317                     * Method not used.
318                     */
319                    public void close() throws SecurityException {
320    
321                    }
322    
323                    /**
324                     * {@inheritDoc}<br>
325                     * Method not used.
326                     */
327                    public void flush() {
328    
329                    }
330    
331                    /**
332                     * {@inheritDoc}<br>
333                     * Places log message into the text area.
334                     */
335                    public void publish(LogRecord record) {
336                            try {
337                                    String msg = getFormatter().format(record);
338                                    addLogMsg(msg);
339                            } catch (Exception ex) {
340                                    log.log(Level.SEVERE, "error logging into text area", ex);
341                                    reportError(null, ex, 0);
342                            }
343                    }
344            }
345    }
